fun unirLenguajes( char **l,  char **m,  var tl,  var tm) 
{
    for (var i = 0; i < tl; i=i+1) {
        print ("%s\n", l);
    }
    for (var i = 0; i < tm; i=i+1) {
        var j;
        for (j = 0; j < tl; j=j+1) {
            if (strcmp(m, l) == 0) {
                break;
            }
        }
        if (j == tl) {
            print ("%s\n", m);
        }
    }
}
